function renderGrid(name) { const grid = document.getElementById('mini-grid'); grid.innerHTML = ""; let oC = 0, pC = 0, activeDays = 0; // activeDays 변수 추가 const itemCount = MEMBERS[name].items.length; for(let i=1; i<=30; i++) { const dS = `2026-04-${String(i).padStart(2,'0')}`; const yoNum = new Date(2026, 3, i).getDay(); const yoColor = (yoNum === 6) ? 'blue' : (yoNum === 0 ? 'red' : '#333'); let dO = 0; let hasData = false; // 해당 날짜에 입력값이 있는지 확인 MEMBERS[name].items.forEach(r => { const s = getS(name, dS, r); if(i <= curD) { if(s !== '대기') hasData = true; // O, X, - 중 하나라도 눌렀다면 데이터 있음 if(s === 'O') oC++; if(s === '-') pC++; } if(s === 'O') dO++; }); // 오늘까지의 날짜 중, '대기'가 아닌 데이터가 있는 날만 분모에 합산 if(i <= curD && hasData) { activeDays++; } const color = dO === itemCount ? '#28C76F' : (dO > 0 ? '#F1C40F' : '#fff'); grid.innerHTML += `
${i}일
(${YOS[yoNum]})
`; } // 🎯 최종 계산: (실천수) / (데이터가 있는 총 항목 수 - 패스 항목 수) const totalPossible = (activeDays * itemCount) - pC; const rate = totalPossible > 0 ? Math.round((oC / totalPossible) * 100) : 0; document.getElementById('rate-summary').innerHTML = `🎯 현재까지 실천 성취율: ${rate}% (패스 ${pC}회 제외)`; }
